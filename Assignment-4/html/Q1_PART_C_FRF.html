%Function
function AR=AmpRatio(mparnew,w)
Kpnew=mparnew(1);
Tau1new=mparnew(2);
Tau2new=mparnew(3);


AR=Kpnew./(sqrt(1+Tau1new^2*w.^2).*sqrt(1+Tau2new^2*w.^2));
end

function Ph=phase(mparnew,w,Dnew)
Kpnew=mparnew(1);
Tau1new=mparnew(2);
Tau2new=mparnew(3);

Ph=-atan(Tau1new.*w)-atan(Tau2new.*w)-Dnew.*w;

end
